54 template swapOpImpl(T: typedesc, op: untyped) =
55 ## We have to use `copyMem` here instead of a simple dereference because they
56 ## may point to a unaligned address. A sufficiently smart compiler _should_
57 ## be able to elide them when they're not necessary.
58 var tmp: T = default(T)
59 copyMem(addr tmp, inp, sizeof(T))
60 tmp = op(tmp)
61 copyMem(outp, addr tmp, sizeof(T))
