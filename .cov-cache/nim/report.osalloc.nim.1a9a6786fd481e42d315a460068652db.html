39 if base == nil: raiseOutOfMem()
40 # memory layout: padding + offset (4 bytes) + user_data
41 # in order to deallocate: read offset at user_data - 4 bytes,
42 # then deallocate user_data - offset
43 let offset = PageSize - (cast[int](base) and (PageSize - 1))
