275 ## Returns cursor position (x, y)
276 ## writes to stdout and expects the terminal to respond via stdin
277 var
278 xStr = ""

280 ch: char = '\0'

282 readX = false

284 # use raw mode to ask terminal for cursor position
285 let fd = getFileHandle(stdin)
286 var oldMode: Termios

290 flushFile(stdout)
291
292 try:
293 # parse response format: [yyy;xxxR
294 while true:
295 let n = readBuffer(stdin, addr ch, 1)
296 if n == 0 or ch == 'R':

302 raise newException(ValueError, "Got unterminated character position message from terminal")

307 xStr.add(ch)

311 # restore previous terminal mode
312 discard fd.tcSetAttr(TCSADRAIN, addr oldMode)
