21 # Generates a case stmt, which assigns the correct enum field given
22 # a normalized string comparison to the `argSym` input.
23 # string normalization is done using passed normalizer.

33 var fVal = ""
34 var fStr = "" # string of current field
35 var fNum = BiggestInt(0) # int value of current field
36 for f in impl:

38 of nnkEmpty: continue # skip first node of `enumTy`
39 of nnkSym, nnkIdent:
40 fVal = f.strVal

110 func span(T: typedesc[HoleyEnum]): int =
111 (T.high.ord - T.low.ord) + 1

113 const invalidSlot = uint8.high

115 proc genLookup[T: typedesc[HoleyEnum]](_: T): auto =
116 const n = span(T)
117 var i = 0
118 assert n <= invalidSlot.int
119 var ret {.noinit.}: array[n, uint8]
120 for ai in mitems(ret): ai = invalidSlot
121 for ai in items(T):
122 ret[ai.ord - T.low.ord] = uint8(i)
123 inc(i)
124 return ret
